``` r
ny_noaa = 
  (ny_noaa %>%
mutate(
    date = as.Date(date),
    year = year(date),
    month = month(date),
    day = day(date)
  ))
```

``` r
ny_noaa =
  ny_noaa %>%
  mutate(
    tmax = ifelse(tmax < -100 | tmax > 120, NA, tmax),
    tmin = ifelse(tmin < -100 | tmin > 120, NA, tmin),
    prcp = ifelse(prcp < 0, NA, prcp),
    snow = ifelse(snow < 0, NA, snow)
  )

data(ny_noaa)
```

``` r
common_snowfall_values = ny_noaa %>%
  filter(snow > 0) %>%
  group_by(snow) %>%
  summarize(count = n()) %>%
  arrange(desc(count))
```
